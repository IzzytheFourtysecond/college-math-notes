\exTwo\ul{Corollary:} If $f: \Delta(0, 1) \to \Delta(0, 1)$ is a biholomorphism with $f(0) = 0$, then $f$ is a rotation.
\begin{myIndent}\exThreeP
	Proof:\\
	Note that $f(0) = 0 \Longrightarrow f^{-1}(0) = 0$. Then by applying Schwarz' lemma to both $f$ and $f^{-1}$, we obtain that $|f(z)| \leq |z|$ and $|f^{-1}(w)| \leq |w|$ for all $z, w \in \Delta(0, 1)$. In particular, by letting $w = f(z)$, we get that $|z| = |f^{-1}(f(z))| \leq |f(z)|$. And hence, we've proven that $|f(z)| = |z|$ for all $z \in \Delta(0, 1)$. By Schwarz' lemma we can conclude that $f(z) = e^{i\alpha}z$ where $\alpha \in \mathbb{R}$. $\blacksquare$\retTwo
\end{myIndent}

\hTwo Question: What if we are given a biholomorphism $f: \Delta \to \Delta$ with $f(0) = a \neq 0$?\retTwo

Idea: Let's find some biholomorphism $\varphi_a : \Delta(0, 1) \to \Delta(0, 1)$ with $\varphi_a(a) = 0$. Then, we must have that the biholomorphism $\varphi_a \circ f$ is a rotation.\retTwo

Specifically, let $\varphi_a(z) = \frac{z - a}{1 - \overline{a}z}$.
\begin{myIndent}\pracTwo
	Observe the following properties:
	\begin{itemize}
		\item $\varphi_a$ extends continuously to $\partial \Delta(0, 1)$. (In fact $\varphi_a$ holomorphically extends to\\ $\mathbb{C} - \{1/\overline{a}\}$. And fortunately $\overline{a} \notin \overline{\Delta}(0, 1)$ since $|a| < 1 \Longrightarrow |1/\overline{a}| > 1$.)\\ [-10pt]
		\item $\varphi_A$ maps $\partial \Delta(0, 1)$ into $\partial \Delta(0, 1)$.
		\begin{myIndent}\pracTwo
			After all, if $|z| = 1$ then we get that:
			
			{\centering$|1 - \overline{a}z| = |1 - a\overline{z}| = |1 - \frac{a}{z}| = |\frac{z - a}{z}| = |z - a|$.\\ [6pt]\par}
		\end{myIndent}

		\item By the maximum modulus property plus the last observation, we also have that\\ $|\varphi_a(z)| \leq 1$ for all $z \in \Delta(0, 1)$.\\ [-10pt]
		
		\item $\varphi_a(0) = \frac{-a}{1} = -a$ and $\varphi_a(a) = \frac{a-a}{1 - \overline{a}a} = \frac{0}{1-|a|^2} = 0$.
		
		\item $\varphi_a$ and $\varphi_{-a}$ are inverses.
		\begin{myIndent}\pracTwo
			Recall the connection between Möbius transformations and matrices acting on $\mathbb{CP}^1$ (see \inLinkRap{Möbius transformations and projective space reference 336}{page 336}). Then note that:

			{\centering $\begin{bmatrix}1 & -a  \\ -\overline{a} & 1\end{bmatrix}\begin{bmatrix}1 & a  \\ \overline{a} & 1\end{bmatrix} = \begin{bmatrix}1 - |a|^2 & 0  \\ 0 & 1 -|a|^2\end{bmatrix} = \begin{bmatrix}1 & a  \\ \overline{a} & 1\end{bmatrix}\begin{bmatrix}1 & -a  \\ -\overline{a} & 1\end{bmatrix}$ \retTwo\par}

			It follows that $\varphi_a \circ \varphi_{-a}$ and $\varphi_{-a} \circ \varphi_a$ correspond to the the identity map on $\mathbb{CP}^1$.\\ [-6pt]
		\end{myIndent}

		\item $\varphi_a^\prime = \frac{(1 - \overline{a}z) + \overline{a}(z - a)}{(1 - \overline{a}z)^2}$. In turn, we can see that $\varphi_a^\prime(0) = 1 - |a|^2$ and $\varphi_a^\prime(a) = \frac{1}{1 - |a|^2}$.\retTwo
	\end{itemize}
\end{myIndent}

\exTwo\ul{Theorem:} If $f: \Delta(0, 1) \to \Delta(0, 1)$ is biholomorphic, then $f(z) = e^{i\theta} \cdot \frac{z - a}{1 - \overline{a}z}$ for some $|a| < 1$ and $\theta \in [0, 2\pi)$.
\begin{myIndent}\exThreeP
	Proof:\\
	Fix $a$ such that $f(a) = 0$. Then let $\widetilde{f} = f \circ \varphi_{-a}$. Note that $\widetilde{f}$ is the composition\\ of two biholomorphisms of $\Delta(0, 1)$. Hence, $\widetilde{f}$ is also a biholomorphism. Also,
	
	{\centering$\widetilde{f}(0) = f(\varphi_{-a}(0)) = f(a) = 0$.\retTwo\par}

	It follows by the prior corollary that $\widetilde{f}(z) = f \circ \varphi_{-a}(z) = e^{i\theta}z$. By composing $\varphi_a$ on the right we thus get that $f(z) = e^{i\theta}\varphi_a(z) = e^{i\theta}\frac{z - a}{1 - \overline{a}z}$. $\blacksquare$\newpage

	\begin{myIndent}\myComment
		Note that on \inLinkRap{Math 220a Conway Exercise III.3.9}{pages 341-342}, I showed that any Möbius transform sending $\partial \Delta(0, 1)$ to itself must have the form $e^{i\theta} \cdot \frac{z - a}{1 - \overline{a}z}$ where $a$ is the point sent to $0$. Also by the \inLinkRap{Math 220a Orientation Principle}{orientation principle (see page 366)}, we have that the prior Möbius transform sends $\Delta(0, 1)$ to $\Delta(0, 1)$ iff $a \in \Delta(0, 1)$.\retTwo

		The significance of this theorem is thus that we proved that \textit{every} biholomorphism from $\Delta(0, 1)$ to itself is a Möbius transform as described in the past homework\\ exercise.\retTwo
	\end{myIndent}
\end{myIndent}

\blab{Set 2 Problem 4 (Blaschke products):} For $a \in \Delta(0, 1) - \{0\}$, define $B_a(z) = \frac{a - z}{1 - \overline{a}z} \cdot \frac{|a|}{a} = -\frac{\overline{a}}{|a|}\varphi_a$.

\begin{enumerate}
	\item[(a)] Let $a \in \Delta(0, 1) - \{0\}$ and $z \in \overline{\Delta}(0, r)$ for $r < 1$. Prove that $\left|\frac{a+|a|z}{(1 - \overline{a}z)a}\right| \leq \frac{1+r}{1-r} $. Then show\\ [-2pt] that on $\overline{\Delta}(0, r)$ we have that $|1 - B_a(z)| \leq \frac{1+r}{1-r}(1 - |a|)$.
	
	\begin{myIndent}\HexOne
		First note that $\frac{a+|a|z}{(1 - \overline{a}z)a} = \frac{(a + |a|z)\frac{\overline{a}}{|a|^2}}{(1 - \overline{a}z)} = \frac{1 + \frac{\overline{a}}{|a|}z}{1 - \overline{a}z}$. Then as $|\overline{a}z| = |\overline{a}||z| \leq r < 1$, we can conclude that:

		{\centering $\left|\frac{a+|a|z}{(1 - \overline{a}z)a}\right| \leq \frac{1 + |\frac{\overline{a}}{|a|}z|}{1 - |\overline{a}z|} \leq \frac{1 + r}{1 - r}$.\retTwo\par}

		Next, note that $1 - B_a(z) = \frac{(1 - \overline{a}z)a}{(1 - \overline{a}z)a} - \frac{(a - z)|a|}{(1 - \overline{a}z)a} = \frac{a-|a|^2z - a|a| + z|a|}{(1 - \overline{a}z)a} = \frac{(a + |a|z)(1 - |a|)}{(1 - \overline{a}z)a}$.\retTwo

		Hence, $|1 - B_a(z)| \leq \frac{1+r}{1-r}(1 - |a|)$ when $z \in \overline{\Delta}(0, r)$.
	\end{myIndent}

	\item[(b)] Let $\{a_n\}_{n \in \mathbb{N}} \subseteq \Delta(0, 1) - \{0\}$ be a sequence of nonzero numbers in the unit disc.\\ [2pt] Assume that $\sum_{n =1}^\infty (1 - |a_n|) < \infty$. Show that the \udefine{Blaschke product} $B(z) = \prod_{n=1}^\infty B_{a_n}(z)$\\ [2pt] converges locally uniformly and absolutely to a holomorphic function\\ [2pt] $B: \Delta(0, 1) \to \Delta(0, 1)$ with zeros only at $a_n$.
	
	\begin{myIndent}\HexOne
		By part (a) we know that $\sum_{n=1}^\infty |1 - B_{a_n}(z)|$ converges absolutely uniformly on\\ compact sets. The other parts are hopefully then obvious.
	\end{myIndent}

	\item[(c)] Show that for $a \in \Delta(0, 1) - \{0\}$, $B_a: \overline{\Delta}(0, 1) \to \overline{\Delta}(0, 1)$ in such a fashion that\\ $|B_a(z)| = 1$ for $|z| = 1$.
	
	\begin{myIndent}\HexOne
		This follows from the the fact $|-\frac{\overline{a}}{|a|}| = 1$ and we already proved this for $\varphi_a$.
	\end{myIndent}

	\item[(d)] Conversely, let $f: \Delta(0, 1) \to \Delta(0, 1)$ be a holomorphic function extending continuously to $\overline{\Delta}(0, 1)$ such that $|f(z)| = 1$ for $|z| = 1$. Show that $f$ can be expressed as a finite Blaschke product:
	
	{\centering $f(z) = cz^m \prod_{n=1}^N B_{a_n}(z)$ (where $|c| = 1$) \retTwo\par}

	\begin{myIndent}\HexOne
		To start off, suppose $f(0) \neq 0$. since $f$ can be extended continuously to $\overline{\Delta}(0, 1)$, we know that $f$ can only have finitely many zeros in $\Delta(0, 1)$ as otherwise the zeros of $f$ would have a limit point in $\partial \Delta$, thus implying $|f(z)| \neq 1$ for all $|z| = 1$.\retTwo

		Now if $a_1, \ldots, a_n$ are the zeros of $f$ (repeated according to multiplicity), let\\ $B(z) = \prod_{j=1}^n B_{a_j}(z)$ and define $g(z) = \frac{f(z)}{B(z)}$. We must have $g$ is holomorphic on\\ [-2pt] $\Delta(0, 1)$, continuous on $\overline{\Delta}(0, 1)$, and nonzero everywhere. Also, $|g(z)| = 1$ when\\ [1pt] $|z| = 1$.\retTwo
		
		By the maximum modulus principle applied to $g$ and $g^{-1}$, we know that $|g(z)| \leq 1$\\ for all $z \in \Delta(0, 1)$ and that $|1/g(z)| \leq 1$ for all $z \in \Delta(0, 1)$. Hence, $|g(z)| = 1$ for all $z \in \Delta(0, 1)$. It follows that $g$ is constant everywhere on $\Delta(0, 1)$. This shows that $f(z) = cB(z)$ where $c$ is some constant. Also as $1 = |f(z)| = |cB(z)| = |c|$ when $|z| = 1$, we have that $|c| = 1$.\retTwo

		Now if $f$ does have a zero of order $m$ at $z = 0$, then we can apply the prior reasoning to get that $\frac{f(z)}{z^m} = cB(z)$. In turn $f(z) = cz^m B(z)$. $\blacksquare$
	\end{myIndent}
\end{enumerate}

\hTwo\ul{Remark:} The action of $\Aut(\Delta(0, 1))$ on $\Delta(0, 1)$ is transitive. After all, for all $a, b \in \Delta(0, 1)$ we have that $f = \varphi_{-b} \circ \varphi_a \in \Aut(\Delta(0, 1))$ satisfies that $f(a) = b$.\retTwo

\exTwo\ul{Application:} Show that if $f: \Delta(0, 1) \to \Delta(0, 1)$ is holomorphic and $f(z) \not\equiv z$, then $f$ has at most $1$ fixed point.
\begin{myIndent}\exThreeP
	Assume $f(a) = a$ and $f(b) = b$ where $a \neq b$. If $a = 0$ then $f(0) = 0 \Longrightarrow f = e^{i\theta}z$ via Schwarz' lemma. But then as $f(b) = b$, we have that $e^{i\theta} = 1$. So, $f \equiv z$ (which we weren't allowing).\retTwo

	Meanwhile if $a \neq 0$, then let $\widetilde{f} = \varphi_a \circ f \circ \varphi_a^{-1}$. But now as $\widetilde{f}$ does have $0$ and $b$ as fixed points. So by our prior reasoning, we know that $\widetilde{f} \equiv z$. Then $f \equiv z$ as well. $\blacksquare$\retTwo
\end{myIndent}

Question: Is there a version of Schwarz' lemma if $f(0) \neq 0$?\retTwo

\exTwo\ul{Proposition:} If $f: \Delta(0, 1) \to \Delta(0, 1)$ is holomorphic, then for all $a \in \Delta(0, 1)$ we have that:

{\centering$\frac{|f^\prime(a)|}{1 - |f(a)|^2} \leq \frac{1}{1 - |a|^2}$.\retTwo\par}

\begin{myIndent}\exThreeP
	\begin{myIndent}\myComment
		Remark:\\
		If $a = 0$, then this proposition is equivalently saying $|f^\prime(a)| \leq 1 - |f(a)|^2$. In\\ particular, if $f(0) = 0$ then this means $|f^\prime(0)| \leq 1$ (which we proved in Schwarz'\\ lemma).\retTwo
	\end{myIndent}

	Proof:\\
	As mentioned in the above remark, we already know this proposition holds when $a = 0$ and $f(a) = 0$. We can use $\Aut(\Delta(0, 1))$ to reduce to this case.\retTwo

	Let $f(a) = \beta$ and set $\widetilde{f} = \varphi_\beta \circ f \circ \varphi_{-a}$. Then $\widetilde{f}(0) = 0$ and $\widetilde{f}$ is a holomorphic function from $\Delta(0, 1)$ to $\Delta(0, 1)$. By Schwarz' lemma, we can conclude that $|\widetilde{f}^\prime(0)| \leq 1$.\retTwo

	Yet also note by chain rule that:
	
	{\centering\begin{tabular}{l}
		$\widetilde{f}^\prime(0) = \varphi_\beta^\prime(f(\varphi_{-a}(0))) \cdot f^\prime(\varphi_{-a}(0)) \cdot \varphi_{-a}^\prime(0)$\\ [4pt]
		$\phantom{\widetilde{f}^\prime(0)} = \varphi_\beta^\prime(\beta) \cdot f^\prime(a) \cdot \varphi_{-a}^\prime(0) = \frac{1}{1 - |\beta|^2} \cdot f^\prime(a) \cdot (1 - |a|^2) = \frac{f^\prime(a)}{1 - |f(a)|^2} \cdot (1 - |a|^2)$
	\end{tabular}\retTwo\par}

	Therefore $\frac{f^\prime(a)}{1 - |f(a)|^2} \cdot (1 - |a|^2) \leq 1 \Longrightarrow \frac{f^\prime(a)}{1 - |f(a)|^2} \leq \frac{1}{1 - |a|^2}$. $\blacksquare$\retTwo

Here is a tangent on why localizations of projective $A$-modules are projective $S^{-1}A$-modules (and a few other things).\retTwo

Suppose $A$ is a unital commutative ring and $S \subseteq A$ is a multiplicatively closed subset. Then localization by $S$ (i.e. the mapping $M \mapsto S^{-1}M$ and $f \mapsto S^{-1}f$) is a well-defined functor from $A$-$\mcateg{mod}$ to $S^{-1}A$-$\mcateg{mod}$.
\begin{myIndent}\pracTwo
	All we need to do is show that $S^{-1}\myId_M = \myId_{S^{-1}M}$ and $S^{-1}(g\circ f) = S^{-1}g \circ S^{-1}f$.\\ Fortunately note that:
	\begin{itemize}
		\item $S^{-1}\myId_M(\frac{m}{s}) = \frac{\myId_M(m)}{s} = \frac{m}{s} = \myId_{S^{-1}M}(\frac{m}{s})$.
		\item $S^{-1}(g \circ f)(\frac{m}{s}) = \frac{g(f(m))}{s} = S^{-1}g(\frac{f(m)}{s}) = S^{-1}g(S^{-1}f(\frac{m}{s}))$.\retTwo
	\end{itemize}
\end{myIndent}

Furthermore, note that localization by $S$ is an exact functor.
\begin{myIndent}\pracTwo
	\ul{Lemma A:} If $M = \{0\}$ then $S^{-1}M = \{0\}$.
	\begin{myIndent}
		Proof:\\
		Suppose $\frac{0}{s} \in S^{-1}M - \{\frac{0}{1}\}$. Then for all $t \in S$ we know $0 = t(1(0) - s(0)) \neq 0$.\\ This is a contradiction.\retTwo
	\end{myIndent}

	\ul{Lemma B:} Suppose $f : M \to M^\prime$ is an $A$-module homomorphism. Then\\ $\ker(S^{-1}f) = S^{-1}\ker(f)$.
	\begin{myIndent}
		Proof:\\
		Note that:
		\begin{itemize}
			\item $\ker(S^{-1}f) = \{\frac{m}{s} \in S^{-1}M : \frac{f(m)}{s} = \frac{0}{s}\}$;
			\item $S^{-1}\ker(f) = S^{-1}\ker(f) = \{\frac{m}{s} \in S^{-1}M : m \in \ker(\phi) \text{ and } s \in S\}$.\retTwo
		\end{itemize}

		Trivially the second set is contained in the first. To see the other inclusion, suppose $\frac{m}{s} \in \ker(S^{-1}f)$. Then $S^{-1}f(\frac{m}{s}) = \frac{f(m)}{s} = \frac{0}{s}$. Hence, we know there exists\\ $t \in A - \mathfrak{m}$ such that $tsf(m) = 0$. In other words $\phi(tsm) = 0$. So, $tsm \in \ker(f)$. And finally, we have that $\frac{m}{s} = \frac{tsm}{ts^2} \in S^{-1}\ker(f)$.\retTwo
	\end{myIndent}

	\ul{Lemma C:} Suppose $f: M \to M^\prime$ is an $A$-module homomorphism. Then\\ $\myIm(S^{-1}f) = S^{-1} \myIm(f)$.
	\begin{myIndent}
		Proof:\\
		Clearly the two sets below are equal:
		\begin{itemize}
			\item $\myIm(S^{-1}f) = \{\frac{f(m)}{s} \in S^{-1}M^\prime : \frac{m}{s} \in S^{-1}M\}$;
			\item $S^{-1}\myIm(f) = \{\frac{f(m)}{s} \in S^{-1}M^\prime : m \in M \text{ and } s \in S\}$.\retTwo
		\end{itemize}
	\end{myIndent}

	Now lemmas B and C show that localization by $S$ sends exact sequences to exact sequences, and lemma A will further show that the functor maps short exact sequences to short exact sequences.\retTwo
\end{myIndent}

One particular consequence of the localization by $S$ being an exact functor is that $S$ it must preserve isomorphisms. In other words, if $M_1 \cong M_2$ as $A$-modules then $S^{-1}M_1 \cong S^{-1}M_2$ as $S^{-1}A$-modules. Before finishing this tangent, we need one more lemma.\newpage

\ul{Lemma:} If $A$ is a unital commutative ring, $S \subseteq A$ is multiplicatively closed, and $\{M_i\}_{i\in I}$ is a collection of $A$ modules, then $S^{-1}(\bigoplus_{i\in I} M_i) \cong \bigoplus_{i\in I} S^{-1}M_i$.

\begin{myIndent}\pracTwo
	Proof:\\
	Define the map $\frac{(x_i)_{i \in I}}{s} \mapsto (\frac{x_i}{s})_{i \in I}$. To see that this is well-defined, suppose $\frac{(x_i)_{i \in I}}{s} = \frac{(y_i)_{i \in I}}{s^\prime}$. Then there exists $t \in S$ with $0 = t(s^\prime(x_i)_{i \in I} - s(y_i)_{i \in I}) = ((t(s^\prime x_i - sy_i))_{i \in I})$. It follows that $\frac{x_i}{s^\prime} = \frac{y_i}{s}$ for all $i \in I$.\retTwo
	
	Hopefully it's obvious why this map is an $S^{-1}A$-module homomorphism.\retTwo
	
	To show that this map is injective, suppose $\frac{(x_i)_{i \in I}}{s} \mapsto (\frac{0}{1})_{i \in I}$. Then for all $i \in I$ we must have that $\frac{x_i}{s} = \frac{0}{1}$, meaning there exists $t_i \in S$ with $t_i(1x_i - 0) = t_ix_i = 0$. That said, as only finitely many of the $x_i$ are nonzero, we can assume without loss of generality that only finitely many of the $t_i$ are not equal to $1$. Hence, we may set $t \coloneqq \prod_{i \in I} t_i$ (where the latter product is just multiplying together the $t_i$ which aren't equal to $1$). Then at last, we have that $t(1(x_i)_{i \in I} - s(0)_{i \in I}) = (0)_{i \in I}$. Hence, we've proven that $\frac{(x_i)_{i \in I}}{s} = \frac{(0)_{i \in I}}{1}$.\retTwo

	Meanwhile, to show our map is surjective, suppose $(\frac{x_i}{s_i})_{i\in I} \in \bigoplus_{i\in I} S^{-1}M_i$. Since without loss of generality we can assume only finitely many $\frac{x_i}{s_i}$ don't equal $\frac{0}{1}$, we can like before consider the products $\prod_{i \in I}s_i$ and $\prod_{j \in I - \{i\}} s_j$. This is important since we then have that:

	{\centering $\frac{((\prod_{j \in I - \{i\}} s_j)x_i)_{i \in I}}{\prod_{i \in I}s_i} \mapsto (\frac{x_i}{s_i})_{i \in I}$ \retTwo\par}
\end{myIndent}

As a consequence of this lemma, suppose $P$ is a projective $A$-module. Then there exists some $A$-module $K$ and some set $I$ such that $P \oplus K \cong \bigoplus_{i \in I}A$. In turn:

{\centering$S^{-1}P \oplus S^{-1}K \cong S^{-1}(P \oplus K) \cong S^{-1}(\bigoplus_{i \in I}A) \cong \bigoplus_{i \in I}S^{-1}A$\retTwo\par}

This shows that $S^{-1}P$ is also a direct summand of a free $S^{-1}A$-module. Hence, $S^{-1}P$ is\\ a projective $S^{-1}A$-module.\retTwo

\begin{enumerate}
	\item[(a)] Prove that every submodule of a finitely generated projective $D$-module is projective if and only if $D$ is a PID.
	\begin{myIndent}\HexOne
		$(\Longleftarrow)$\\
		Suppose $D$ is a PID and consider any finitely generated projective $D$-module $M$. Then we know that $M$ is free (see \inLinkRap{Handy Diagram page 665}{page 665}). In turn any submodule of $M$ is also free (since $D$ is a PID) and thus projective.\retTwo

		$(\Longrightarrow)$\\
		Suppose $D$ is not a PID and consider any finitely generated projective $D$-module $M$. By problem 3c on the previous 200b problem set (see \inLinkRap{Math 200b Set 5 Problem 3c}{pages 658-659}), we know that $D$ being local and $M$ being finitely generated and projective implies that $M$ is a free module. Yet as $D$ is not a PID, we know that there is some submodule $N \subseteq M$ that is not a free module (see \inLinkRap{Math 200b Set 5 Problem 2}{set 5 problem 3 on page 657}). Therefore, if $N$ were projective we'd have to have that $N$ is not finitely generated. After all if $N$ were finitely generated and projective we'd know by problem 3c that $N$ is a free module.\retTwo
		
		However, $M$ being finitely generated and $D$ being Noetherian implies that $M$ is a Noetherian module. (see set 2 problem 6 on \inLinkRap{Math 200b Set 2 Problem 6}{pages 570-572}). It follows that $N \subseteq M$ is finitely generated. In turn, we know $N$ is not projective.\newpage
	\end{myIndent}

	\item[(b)] Find a local Noetherian integral domain which is not a PID.
	\begin{myIndent}\HexOne
		Consider the polynomial ring $\mathbb{Q}[x, y]$. This is a Noetherian integral domain which is not a PID. Also, $\mathbb{Q}[x, y]$ is local. After all, let $\mathfrak{m} = \langle x, y\rangle$. Then $\mathfrak{m}$ is the ideal containing all non-constant polynomials. Furthermore, every element not in $\mathfrak{m}$ is a unit. It follows that $\Max(\mathbb{Q}[x, y]) = \{\mathfrak{m}\}$ (see \inLinkRap{Math 200b Set 4 Problem 6}{set 4 problem 6 on page 626})and therefore $\mathbb{Q}[x, y]$ is a local ring.
	\end{myIndent}

	\item[(c)] Show that a submodule of a finitely generated projective module is not necessarily projective.  
	\begin{myIndent}\HexOne

Hence, we can conclude that $\langle x, y\rangle$ is \textit{not} a projective $\mathbb{Q}[x, y]$-module. $\blacksquare$

	Write $\beta_i = \frac{r_i}{s_i} \coloneqq S^{-1}g_i(1)$. Then $g_i(x) = S^{-1}g_i(\frac{x}{1}) = \frac{r_ix}{s_i}$ for all $x \in \mathfrak{a}$. Hence, each\\ [-1pt] $g_i : \mathfrak{a} \to D$ can be written in the form $g_i(x) = \frac{r_i}{s_i} x$ for some $\beta_i = \frac{r_i}{s_i} \in F$.\retTwo

	Finally, $g(x) = (\beta_1 x, \ldots, \beta_n x)$. Since $f \circ g = \myId_{\mathfrak{a}}$, we know that:
	
	{\center$x \cdot \sum_{i=1}^n \beta_i a_i = \sum_{i=1}^n (\beta_i x) a_i = f(\beta_1 x, \ldots, \beta_n x) = x$ for all $x \in \mathfrak{a}$.\retTwo\par}

	In particular, since $\mathfrak{a} \neq \{0\}$ we can choose some nonzero $x \in \mathfrak{a}$. Then after canceling out $x$ we get that $\beta_1 a_1 + \cdots + \beta_n a_n = 1$.\retTwo

	Define $M = D\beta_1 + \cdots + D\beta_n$ be a $D$-submodule of $F$. It is easy to see that\\ $\mathfrak{a}M = \langle a_j \beta_i: i, j \in \{1, \ldots, n\} \rangle$. But as $a_j \beta_i = g_i(a_j) \in D$ for all $j$ and $i$, we know\\ that $\mathfrak{a}M \subseteq D$. Also, we showed above that $1 \in \mathfrak{a}M$. So, $\mathfrak{a}M = D$.\retTwo

	$(\Longleftarrow)$\\
	Suppose $\mathfrak{a}M = D$ for some $D$-submodule $M$ of $F$. Then:
	
	{\centering$1 = a_1 \beta_1 + \cdots + a_n \beta_n$ for some $a_i \in \mathfrak{a}$ and $\beta_i \in M$.\retTwo\par}

	Also by setting some $\beta_i = 0$ if necessary, we can assume $\mathfrak{a} = \langle a_1, \ldots, a_n\rangle$.\retTwo

	Consider the standard SES given by {\color{red}$(*)$}. Then define $g: \mathfrak{a} \to D^n$ by:
	
	{\centering$g(x) = (\beta_1x, \ldots, \beta_n x)$.\retTwo\par}

	Note that $g$ is well-defined as $\beta_i \mathfrak{a} \subseteq D$ for all $i$ (since $\mathfrak{a}M = D$). Also $g$ is clearly a\\ $D$-module homomorphism. And thirdly, $f(g(x)) = \sum_{i=1}^n a_i\beta_i x = 1x$ for all $x \in \mathfrak{a}$. It follows that {\color{red}$(*)$} is a splitting S.E.S. So, we may write $D^n \cong \ker(f) \oplus \mathfrak{a}$ and this proves that $\mathfrak{a}$ is projective. $\blacksquare$\retTwo
\end{myIndent}
